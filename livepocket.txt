// ==UserScript==
// @name         Stay插件-零延迟格局无损姓名切分
// @namespace    http://tampermonkey.net
// @version      10.0
// @description  毫秒级同步拦截：解决打开网页时的位置移动和闪烁问题，100%保持原有字体格局。
// @author       AI助手
// @match        *://*/*
// @grant        none
// @run-at       document-start
// ==/UserScript==

(function() {
    'use strict';

    // 核心处理函数
    function doNameSplit(container) {
        // 防止对同一个标签重复处理导致死循环
        if (container.hasAttribute('data-stay-split-done')) return;

        const titles = container.querySelectorAll('.my-ticket-info__name-title');
        if (titles.length < 2) return;

        // 1. 隐藏原网页右边错位的“名”标签（用透明度，不破坏CSS选择器格局）
        titles[1].style.opacity = '0';
        titles[1].style.position = 'absolute';
        titles[1].style.pointerEvents = 'none';

        // 2. 提取并清洗名字纯文本
        let rawText = "";
        container.childNodes.forEach(node => {
            if (node.nodeType === Node.TEXT_NODE) {
                rawText += node.textContent;
            }
        });
        let cleanText = rawText.replace(/[\ufffb\u200b-\u200d\ufeff]/g, '').trim();

        // 3. 严格寻找空格切分
        let spaceIndex = cleanText.search(/[\s　]+/);
        if (spaceIndex !== -1) {
            let lastName = cleanText.substring(0, spaceIndex).trim();
            let firstName = cleanText.substring(spaceIndex + 1).replace(/^[\s　]+/, '').trim();

            if (lastName && firstName) {
                // 标记已处理
                container.setAttribute('data-stay-split-done', 'true');

                // 4. 精准替换文本节点，完全不触碰外层原生标签
                container.childNodes.forEach(node => {
                    if (node.nodeType === Node.TEXT_NODE) {
                        const nameSpan = document.createElement('span');
                        // 注入原生class，确保字体、颜色100%继承
                        nameSpan.innerHTML = `${lastName} <span class="my-ticket-info__name-title">名</span> ${firstName}`;
                        node.parentNode.replaceChild(nameSpan, node);
                    }
                });
            }
        }
    }

    // 【零延迟核心】：使用底层的 MutationObserver 动态监听 DOM 树的构建
    const observer = new MutationObserver((mutations, obs) => {
        const container = document.querySelector('.my-ticket-info__name');
        if (container) {
            // 只要一在内存中发现该元素，立刻执行，此时网页还没渲染到屏幕上
            doNameSplit(container);
            // 处理完成后直接断开监听，释放手机内存
            if (container.hasAttribute('data-stay-split-done')) {
                obs.disconnect();
            }
        }
    });

    // 从网页刚开始加载的那一刻起，就启动最高级别的肉眼不可见监控
    observer.observe(document.documentElement, {
        childList: true,
        subtree: true
    });

    // 保底机制：如果因为特殊原因漏掉了，在页面完全加载后再触发一次
    window.addEventListener('DOMContentLoaded', () => {
        const container = document.querySelector('.my-ticket-info__name');
        if (container) doNameSplit(container);
    });

})();
