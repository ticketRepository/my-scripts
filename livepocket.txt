// ==UserScript==
// @name         livepocket 姓名修改
// @namespace    lanlan
// @version      4.1
// @description  最终版。
// @author       lanlan
// @match        https://t.livepocket.jp/*
// @match        https://livepocket.jp/*
// @grant        none
// @run-at       document-start
// ==/UserScript==

(function () {
  'use strict';

  let observer = null;

  function doNameSplit(container) {
    if (container.hasAttribute('data-stay-split-done')) return;

    const cleanText = container.textContent
      .replace(/[\ufffb\u200b-\u200d\ufeff]/g, '')
      .replace(/氏|名/g, '')
      .trim();

    const spaceIndex = cleanText.search(/\s+/);
    if (spaceIndex === -1) return;

    const lastName = cleanText.substring(0, spaceIndex);
    const firstName = cleanText.substring(spaceIndex).trim();
    if (!lastName || !firstName) return;

    container.setAttribute('data-stay-split-done', 'true');
    container.innerHTML =
      '<span class="my-ticket-info__name-title">氏</span>' + lastName +
      '<span class="my-ticket-info__name-title">名</span>' + firstName;
  }

  function processAll() {
    document.querySelectorAll('.my-ticket-info__name').forEach(doNameSplit);
  }

  function tryStop() {
    // 所有实例都处理完了才 disconnect
    const all = document.querySelectorAll('.my-ticket-info__name');
    const allDone = all.length > 0 && [...all].every(el => el.hasAttribute('data-stay-split-done'));
    if (allDone && observer) {
      observer.disconnect();
      observer = null;
    }
  }

  observer = new MutationObserver(() => {
    processAll();
    tryStop();
  });
  observer.observe(document.documentElement, { childList: true, subtree: true });

  window.addEventListener('DOMContentLoaded', () => {
    processAll();
    tryStop();
  });
})();
