// ==UserScript==
// @name         万代专用
// @namespace    lanlan
// @version      18.0
// @description  手机本地纯视觉自嗨！全自动动态抓取会员信息页的任意最新汉字名，并完美无缝替换全站所有历史订单里的旧名与假名
// @match        *://*/*
// @grant        none
// @run-at       document-start
// ==/UserScript==

(function() {
    'use strict';

    // === 1. 初始化动态名字变量（优先从浏览器本地存储读取，实现跨页面同步） ===
    let currentDynamicName = localStorage.getItem('STAY_DYNAMIC_VISUAL_NAME') || '';

    // === 2. 自动抓取器：在会员信息页悄悄偷看并记住你当前的最新姓名 ===
    function autoSniffLatestName() {
        const labels = document.querySelectorAll('.block-mypage-member-info-label');
        labels.forEach(label => {
            // 精准定位到“氏名（漢字）”对应的那个值
            if (label.textContent.trim() === '氏名（漢字）' || label.textContent.includes('氏名')) {
                const dlItem = label.closest('.block-mypage-member-info-item') || label.parentElement;
                const valueNode = dlItem ? dlItem.querySelector('.block-mypage-member-info-value') : null;
                
                if (valueNode && valueNode.textContent.trim()) {
                    const detectedName = valueNode.textContent.trim();
                    // 如果抓到的名字有效，且跟之前记住的不一样，立刻更新记忆库
                    if (detectedName && currentDynamicName !== detectedName) {
                        currentDynamicName = detectedName;
                        localStorage.setItem('STAY_DYNAMIC_VISUAL_NAME', currentDynamicName);
                    }
                }
            }
        });
    }

    // === 3. 核心大招：历史订单页面的“全自动纯视觉大清洗” ===
    function cleanHistoryAndStaticText() {
        // 如果还没有在会员页成功抓取到任何名字，先不盲目替换，避免显示空白
        if (!currentDynamicName) return;

        // 轨迹 A：根据你提供的 HTML 结构，直接对订单详情大盒子实施源码层面的“降维打击”
        const contents = document.querySelectorAll('.block-mypage-history-block-detail-contents-block-content');
        contents.forEach(content => {
            let html = content.innerHTML;
            
            // 判定该盒子里是否包含了订单里的旧人名。
            // 只要第一行文本在换行符 <br> 前面包含了非地址类的日文字符，且不是支付方式，就判定需要替换
            if (html.includes('<br') || html.includes('<BR')) {
                // 精准利用正则：将第一个换行符 <br> 之前的一整行旧名字、旧假名（不管是什么名字）统统抹去，换成你改好的最新动态名字
                // 这能完美保留下面的邮编、地址、电话，且抹除包括括号假名在内的全部垃圾文字
                const lines = html.split(/<br>|<BR>/);
                if (lines[0] && lines[0].trim() !== currentDynamicName && !lines[0].includes('结算') && !lines[0].includes('決済')) {
                    lines[0] = currentDynamicName;
                    content.innerHTML = lines.join('<br>');
                }
            } else {
                // 如果是没有换行符的“ご依頼主（委托人）”盒子，只要里面的字不是你最新的名字，且不是支付方式，直接强行重写
                const textTrim = content.textContent.trim();
                if (textTrim && textTrim !== currentDynamicName && !textTrim.includes('结算') && !textTrim.includes('決済') && textTrim.length < 20) {
                    content.innerHTML = currentDynamicName;
                }
            }
        });

        // 轨迹 B：顺便处理“氏名（カナ）”那栏锁死的旧假名（藤原、丹羽等对应假名全杀）
        const memberInfoLabels = document.querySelectorAll('.block-mypage-member-info-label');
        memberInfoLabels.forEach(label => {
            if (label.textContent.trim() === '氏名（カナ）') {
                const dlItem = label.closest('.block-mypage-member-info-item') || label.parentElement;
                const valueNode = dlItem ? dlItem.querySelector('.block-mypage-member-info-value') : null;
                if (valueNode && valueNode.textContent.trim() !== currentDynamicName) {
                    valueNode.textContent = currentDynamicName; // 把露馅的假名也无脑洗成你最新的汉字名
                }
            }
        });
    }

    // === 4. 手机端全周期高频突变监控 ===
    function engineRun() {
        autoSniffLatestName();         // 随时抓取新名字
        cleanHistoryAndStaticText();   // 随时强洗旧名字
    }

    engineRun();
    window.addEventListener('DOMContentLoaded', engineRun);
    window.addEventListener('load', engineRun);

    // 每 35 毫秒（一秒钟疯狂高频洗屏近 30 次）死循环压制
    // 现代前端框架无论何时把旧订单数据刷出来，都会在 0.03 秒内被 Stay 插件瞬间掉包
    setInterval(engineRun, 35);

    // 突变保镖双重监控，彻底杜绝反弹
    const observer = new MutationObserver(engineRun);
    observer.observe(document.documentElement, { 
        childList: true, 
        subtree: true, 
        characterData: true 
    });

})();
