// ==UserScript==
// @name         万代全能匹配
// @namespace    lanlan
// @version      2.0
// @description 
// @match        *://*/*
// @grant        none
// @run-at       document-end
// ==/UserScript==

(function() {
    'use strict';

    let myCustomName = localStorage.getItem('STAY_CUSTOM_VISUAL_NAME') || '';

    function convertRomajiToKatakana(romajiText) {
        if (!romajiText) return "";
        let text = romajiText.toUpperCase().trim();
        if (text === "DONG MENGYUAN" || text === "DONGMENGYUAN") return "ドン メンユアン";
        if (text === "YAMAMOTO HANYU" || text === "YAMAMOTOHANYU") return "ヤマモト ハニュウ";

        const mapping = {
            'CHENG': 'チェン', 'LIU': 'リュウ', 'WANG': 'ワン', 'ZHANG': 'ジャン', 'YANG': 'ヤン', 'HUANG': 'ファン',
            'ZHAO': 'ジャオ', 'ZHOU': 'ジョウ', 'WU': 'ウー', 'XU': 'シュー', 'SUN': 'スン', 'GAO': 'ガオ', 'LIN': 'リン',
            'MENG': 'メン', 'YUAN': 'ユアン', 'DONG': 'ドン', 'A': 'ア', 'I': 'イ', 'U': 'ウ', 'E': 'エ', 'O': 'オ'
        };

        let words = text.split(/[\s　]+/);
        let resultWords = words.map(word => {
            let kana = ""; let i = 0;
            while (i < word.length) {
                let matched = false;
                for (let len = 4; len > 0; len--) {
                    if (i + len <= word.length) {
                        let substr = word.substr(i, len);
                        if (mapping[substr]) { kana += mapping[substr]; i += len; matched = true; break; }
                    }
                }
                if (!matched) i++;
            }
            return kana;
        });
        return resultWords.filter(w => w.length > 0).join(' ');
    }

    let currentKatakanaName = convertRomajiToKatakana(myCustomName);

    // === 清洗函数 ===
    function cleanWebPageVisuals() {
        if (!myCustomName) return;

        // 【优化项 1】：使用极速选择器直接锁定
        const contents = document.querySelectorAll('.block-mypage-history-block-detail-contents-block-content');
        contents.forEach(content => {
            let html = content.innerHTML;
            
            if (html.includes('<br') || html.includes('<BR')) {
                const lines = html.split(/<br>|<BR>/i);
                if (lines && lines.length > 0) {
                    const firstLineText = lines[0].trim();
                    const expectedLine = myCustomName + "(" + currentKatakanaName + ")";
                    // 【优化项 2】：增加缓存防线，只有对不上的时候才准修改DOM，防止死循环重绘
                    if (firstLineText !== myCustomName && firstLineText !== expectedLine && !html.includes('決済') && !html.includes('结算')) {
                        lines[0] = expectedLine; 
                        content.innerHTML = lines.join('<br>');
                    }
                }
            } else {
                const textTrim = content.textContent.trim();
                if (textTrim && textTrim !== myCustomName && !textTrim.includes('決済') && !textTrim.includes('结算') && textTrim.length < 30) {
                    content.innerHTML = myCustomName;
                }
            }
        });

        // 会员页面的精准更新
        const labels = document.querySelectorAll('.block-mypage-member-info-label');
        labels.forEach(label => {
            const labelText = label.textContent.trim();
            if (labelText === '氏名（漢字）' || labelText === '氏名（カナ）') {
                const dlItem = label.closest('.block-mypage-member-info-item') || label.parentElement;
                const valueNode = dlItem ? dlItem.querySelector('.block-mypage-member-info-value') : null;

                if (valueNode) {
                    if (labelText === '氏名（漢字）' && valueNode.textContent.trim() !== myCustomName) {
                        valueNode.textContent = myCustomName;
                    }
                    if (labelText === '氏名（カナ）' && valueNode.textContent.trim() !== currentKatakanaName) {
                        valueNode.textContent = currentKatakanaName;
                    }
                }
            }
        });
    }

    // === 【优化项 3】：防抖节流阀（Debounce）===
    // DOM优化
    let debounceTimer = null;
    function debounceWash() {
        clearTimeout(debounceTimer);
        debounceTimer = setTimeout(cleanWebPageVisuals, 20);
    }

    // 双击触发
    document.addEventListener('dblclick', function(event) {
        if (event.target.textContent.includes('会員情報') || event.target.textContent.includes('会员')) {
            const userInput = prompt("输入英文名：", myCustomName || "DONG MENGYUAN");
            if (userInput !== null) {
                const trimmedInput = userInput.trim();
                if (trimmedInput) {
                    myCustomName = trimmedInput;
                    currentKatakanaName = convertRomajiToKatakana(myCustomName);
                    localStorage.setItem('STAY_CUSTOM_VISUAL_NAME', myCustomName);
                    cleanWebPageVisuals();
                }
            }
        }
    });

    // === 启动轻量监控 ===
    // 防抖
    window.addEventListener('DOMContentLoaded', cleanWebPageVisuals);
    window.addEventListener('load', cleanWebPageVisuals);

    const observer = new MutationObserver(debounceWash);
    observer.observe(document.documentElement, { 
        childList: true, 
        subtree: true, 
        characterData: true 
    });

    cleanWebPageVisuals();
})();
