// ==UserScript==
// @name         姓名同步 
// @namespace    lanlan
// @version      10.0
// @description  自动读取修改页最新姓名(如DONG MENGYUAN)，全自动智能替换历史订单中的丹羽信二及假名
// @match        *://*/*
// @grant        none
// @run-at       document-end
// ==/UserScript==

(function() {
    'use strict';

    // === 1. 初始化姓名读取（优先从本地记忆中取，如果没有则默认使用你改好的名字） ===
    let currentTargetLastName = localStorage.getItem('STAY_NEW_LAST_NAME') || 'DONG'; 
    let currentTargetFirstName = localStorage.getItem('STAY_NEW_FIRST_NAME') || 'MENGYUAN'; 

    // === 核心功能 A：自动偷看并记住修改页的最新真实姓名 ===
    function autoSniffNameFromProfile() {
        // 尝试寻找修改页的【名】隐藏域
        const hiddenInput = document.getElementById('F_NAME') || document.querySelector('input[name="F_NAME"]');
        // 尝试寻找【姓】的输入框（通过常见关键词识别）
        let lastNameInput = null;
        const inputs = document.querySelectorAll('input');
        inputs.forEach(input => {
            const nameAttr = (input.name || '').toLowerCase();
            const idAttr = (input.id || '').toLowerCase();
            if ((nameAttr.includes('last') || idAttr.includes('last') || input.placeholder?.includes('姓')) && input.type !== 'hidden') {
                lastNameInput = input;
            }
        });

        // 如果在修改页面，且找到了对应的元素，自动把当前的真实值保存到记忆库
        if (hiddenInput && hiddenInput.value && hiddenInput.value !== '信二') {
            if (currentTargetFirstName !== hiddenInput.value) {
                currentTargetFirstName = hiddenInput.value;
                localStorage.setItem('STAY_NEW_FIRST_NAME', currentTargetFirstName);
            }
        }
        if (lastNameInput && lastNameInput.value && lastNameInput.value !== '丹羽' && !lastNameInput.value.includes(' ')) {
            if (currentTargetLastName !== lastNameInput.value) {
                currentTargetLastName = lastNameInput.value;
                localStorage.setItem('STAY_NEW_LAST_NAME', currentTargetLastName);
            }
        }
    }

    // === 核心功能 B：历史订单页面的“全自动文本替换” ===
    function cleanHistoryAndStaticText() {
        const walker = document.createTreeWalker(document.body, NodeFilter.SHOW_TEXT, null, false);
        let node;
        
        // 智能拼接新姓名格式（例如：DONG MENGYUAN 或 DONGMENGYUAN）
        const fullNameTargetWithSpace = currentTargetLastName + ' ' + currentTargetFirstName;
        const fullNameTarget = currentTargetLastName + currentTargetFirstName;

        while (node = walker.nextNode()) {
            let text = node.nodeValue;

            // 1. 【核心】自动干掉“丹羽 信二(ニワ シンジ)”一整串，替换为最新的 DONG MENGYUAN
            if (text.includes('丹羽 信二') || text.includes('丹羽信二')) {
                node.nodeValue = text.replace(/丹羽\s*信二\s*\([^)]*\)/g, fullNameTargetWithSpace)
                                     .replace(/丹羽\s*信二/g, fullNameTargetWithSpace)
                                     .replace(/丹羽信二/g, fullNameTarget);
            }

            // 2. 局部单字兜底修正
            if (text.trim() === '信二') {
                node.nodeValue = text.replace(/信二/g, currentTargetFirstName);
            }
            if (text.trim() === '丹羽') {
                node.nodeValue = text.replace(/丹羽/g, currentTargetLastName);
            }
        }
    }

    // === 核心功能 C：如果你以后还要用空格改名字，保留空格切分逻辑（防止姓名名Bug） ===
    function handleLiveInput(event) {
        const currentInput = event.target;
        if (currentInput.tagName === 'INPUT' && currentInput.name !== 'F_NAME' && currentInput.type !== 'hidden') {
            const rawValue = currentInput.value;
            if (rawValue.includes(' ') || rawValue.includes('　')) {
                const parts = rawValue.split(/[\s　]+/);
                currentTargetLastName = parts[0];
                currentTargetFirstName = parts.slice(1).join(' ');

                localStorage.setItem('STAY_NEW_LAST_NAME', currentTargetLastName);
                localStorage.setItem('STAY_NEW_FIRST_NAME', currentTargetFirstName);

                if (currentInput.value !== currentTargetLastName) {
                    currentInput.value = currentTargetLastName;
                    currentInput.dispatchEvent(new Event('input', { bubbles: true }));
                }

                // 同步更新修改页的隐藏域
                const hiddenInput = document.getElementById('F_NAME') || document.querySelector('input[name="F_NAME"]');
                if (hiddenInput) {
                    hiddenInput.value = currentTargetFirstName;
                    hiddenInput.dispatchEvent(new Event('input', { bubbles: true }));
                    const parentDiv = hiddenInput.closest('.form-input') || hiddenInput.parentElement;
                    const spanText = parentDiv ? parentDiv.querySelector('span.text') : null;
                    if (spanText) spanText.textContent = currentTargetFirstName;
                }
                cleanHistoryAndStaticText();
            }
        }
    }

    // === 4. 监控与运行 ===
    document.addEventListener('input', handleLiveInput, true);

    const observer = new MutationObserver(() => {
        autoSniffNameFromProfile(); // 持续偷看修改页的真实姓名
        cleanHistoryAndStaticText(); // 持续守护历史订单页文本
    });

    observer.observe(document.body, {
        childList: true,
        subtree: true,
        characterData: true
    });

    // 页面加载完成后立刻执行
    autoSniffNameFromProfile();
    cleanHistoryAndStaticText();
})();
